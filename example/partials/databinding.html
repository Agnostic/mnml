Databinding

app.controller('databinding', function($name){
  var self      = this;
  this.name     = $name || 'World';
  this.boolean  = true;
  this.radio    = 2;
  this.select   = 2;
  this.textarea = "Hello Textarea";

  watch(self, function(){
    console.log('Watch', self.name);
  });

  app.fireEvent('setPage', 'databinding');
});
